// ============================================================ // SMART HOSTEL ROOM ALLOCATION — app.js // 0/1 Knapsack DP (exact port of C algorithm) // ============================================================ // ── State ────────────────────────────────────────────────── let students = []; // { id, name, gender, year, distanceKm, cgpa, hasMedical, hasFinancial, weight, priorityScore } let nextId = 1; let lastDP = { male: null, female: null }; let animTimer = null; // ── Professional Scoring (max 100 pts) ──────────────────── function computePriority(s) { let score = 0; // 1. Medical / Disability (max 20 pts) if (s.hasMedical) score += 20; // 4. Financial Need — EWS/BPL/Scholarship (max 15 pts) if (s.hasFinancial) score += 15; // 5. Academic Merit — CGPA (max 5 pts) if (s.cgpa >= 8.5) score += 5; else if (s.cgpa >= 7.0) score += 3; else score += 1; return score; } function scoreBreakdown(s) { let dist = 0; if (s.distanceKm >= 200) dist = 40; else if (s.distanceKm >= 150) dist = 35; else if (s.distanceKm >= 100) dist = 30; else if (s.distanceKm >= 50) dist = 20; else dist = 10; const yr = s.year === 1 ? 20 : s.year === 2 ? 15 : s.year === 3 ? 10 : 5; const med = s.hasMedical ? 20 : 0; const fin = s.hasFinancial ? 15 : 0; const cgpa = s.cgpa >= 8.5 ? 5 : s.cgpa >= 7.0 ? 3 : 1; return { dist, yr, med, fin, cgpa }; } // ── 0/1 Knapsack DP (exact C logic) ────────────────────── function knapsack(group, W) { const n = group.length; const dp = Array.from({ length: n + 1 }, () => new Array(W + 1).fill(0)); for (let i = 1; i <= n; i++) { const wi = group[i - 1].weight; // always 1 const vi = group[i - 1].priorityScore; for (let w = 0; w <= W; w++) { dp[i][w] = dp[i - 1][w]; // exclude if (wi <= w) { const inc = vi + dp[i - 1][w - wi]; if (inc > dp[i][w]) dp[i][w] = inc; // include } } } return dp; } // ── Backtrack ───────────────────────────────────────────── function backtrack(group, W, dp) { const n = group.length; let w = W; const allocated = []; const btCells = []; for (let i = n; i >= 1; i--) { if (dp[i][w] !== dp[i - 1][w]) { allocated.push(group[i - 1]); btCells.push({ i, w }); w -= group[i - 1].weight; } } return { allocated, btCells }; } // ── INIT ────────────────────────────────────────────────── window.addEventListener('DOMContentLoaded', () => { // Slider listeners ['male', 'female'].forEach(g => { const sl = document.getElementById(`cap-${g}`); const out = document.getElementById(`cap-${g}-val`); sl.addEventListener('input', () => { out.value = sl.value; }); }); // Toggle labels document.getElementById('f-medical').addEventListener('change', e => { document.getElementById('medical-label').textContent = e.target.checked ? } // ── ADD STUDENT ─────────────────────────────────────────── function addStudent(e) { e.preventDefault(); const name = document.getElementById('f-name').value.trim(); const gender = document.querySelector('input[name="gender"]:checked')?.value;
